# PHASE I EXPERIMENTOS (M1-M4) - BLOCO UNICO PYTHON (COLAB)
import subprocess
import pathlib

root = pathlib.Path('/content')
repo = root / 'oars-colab-mvp'

if repo.exists():
    subprocess.run('git pull', shell=True, cwd=repo, check=True)
else:
    subprocess.run('git clone https://github.com/PAMF2/oars-colab-mvp.git', shell=True, cwd=root, check=True)

subprocess.run('pip -q install -e .[dev] datasets huggingface_hub', shell=True, cwd=repo, check=True)

# baixar miniF2F real
subprocess.run(
    "python - << 'PY'\n"
    "from huggingface_hub import hf_hub_download\n"
    "import shutil\n"
    "src = hf_hub_download(repo_id='Tonic/MiniF2F', repo_type='dataset', filename='minif2f.jsonl')\n"
    "shutil.copy(src, 'minif2f_raw.jsonl')\n"
    "print('saved minif2f_raw.jsonl')\n"
    "PY",
    shell=True,
    cwd=repo,
    check=True,
)

# rodar milestones M1-M4
subprocess.run(
    'python scripts/run_phase1.py --raw minif2f_raw.jsonl --milestones M1,M2,M3,M4 --kfold 5 --seeds 10 --epochs 20 --output-dir outputs/phase1',
    shell=True,
    cwd=repo,
    check=True,
)

# mostrar relatorio final
subprocess.run('cat outputs/phase1/phase1_report.json', shell=True, cwd=repo, check=True)
