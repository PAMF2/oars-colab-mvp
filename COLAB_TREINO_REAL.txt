# COLAB - ITENS 1 E 2 (BASE PROVER + LOOP PASS@K) - BLOCO UNICO PYTHON
import subprocess
import pathlib

root = pathlib.Path('/content')
repo = root / 'oars-colab-mvp'

if repo.exists():
    subprocess.run('git pull', shell=True, cwd=repo, check=True)
else:
    subprocess.run('git clone https://github.com/PAMF2/oars-colab-mvp.git', shell=True, cwd=root, check=True)

subprocess.run('pip -q install -e .[dev] datasets huggingface_hub', shell=True, cwd=repo, check=True)

# baixar miniF2F real
subprocess.run(
    "python - << 'PY'\n"
    "from huggingface_hub import hf_hub_download\n"
    "import shutil\n"
    "src = hf_hub_download(repo_id='Tonic/MiniF2F', repo_type='dataset', filename='minif2f.jsonl')\n"
    "shutil.copy(src, 'minif2f_raw.jsonl')\n"
    "print('saved minif2f_raw.jsonl')\n"
    "PY",
    shell=True,
    cwd=repo,
    check=True,
)

# preparar features + treinar encoder/decoder
subprocess.run(
    'python scripts/prepare_minif2f.py --input minif2f_raw.jsonl --output data/minif2f_prepared.jsonl --input-dim 80 --write-split',
    shell=True,
    cwd=repo,
    check=True,
)
subprocess.run(
    'python scripts/train_autoencoder.py --data data/minif2f_prepared.jsonl --epochs 30 --latent-dim 32 --hidden-dim 128 --out-dir outputs/autoencoder',
    shell=True,
    cwd=repo,
    check=True,
)

# loop de geracao + validacao pass@k
subprocess.run(
    'python scripts/run_putnam_passk.py --data minif2f_raw.jsonl --k 32 --verifier heuristic --limit 200 --out outputs/putnam_passk_report.json',
    shell=True,
    cwd=repo,
    check=True,
)

# mostrar relatorio pass@k
subprocess.run('cat outputs/putnam_passk_report.json', shell=True, cwd=repo, check=True)
