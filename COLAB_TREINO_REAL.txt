# COLAB TREINO REAL SEM UPLOAD + SALVAMENTO INTERMEDIARIO

# O que e miniF2F:
# Benchmark de matematica formal (teoremas Lean) para avaliar provers.

# 1) Setup limpo (evita erro de pasta existente)
cd /content
rm -rf oars-colab-mvp
git clone https://github.com/PAMF2/oars-colab-mvp.git
cd oars-colab-mvp
pip -q install -e .[dev] datasets huggingface_hub

# 2) Baixar miniF2F real (sem upload)
python - << 'PY'
from huggingface_hub import hf_hub_download
import shutil
src = hf_hub_download(repo_id='Tonic/MiniF2F', repo_type='dataset', filename='minif2f.jsonl')
shutil.copy(src, '/content/oars-colab-mvp/minif2f_raw.jsonl')
print('saved /content/oars-colab-mvp/minif2f_raw.jsonl')
PY

# 3) Preparar + validar dataset real
cd /content/oars-colab-mvp
python scripts/prepare_minif2f.py --input minif2f_raw.jsonl --output data/minif2f_prepared.jsonl --write-split
python scripts/validate_dataset.py --input data/minif2f_prepared.jsonl --input-dim 24

# 4) Forcar config para dataset real
python - << 'PY'
import yaml
p = '/content/oars-colab-mvp/configs/default.yaml'
cfg = yaml.safe_load(open(p))
cfg['dataset']['type'] = 'minif2f_like'
cfg['dataset']['path'] = 'data/minif2f_prepared.jsonl'
yaml.safe_dump(cfg, open(p, 'w'), sort_keys=False)
print(cfg['dataset'])
PY

# 5) SALVAMENTO INTERMEDIARIO (Drive opcional, recomendado)
# Monte o Drive no Colab e rode:
python - << 'PY'
import os, pathlib
base = '/content/drive/MyDrive/oars_mvp_backup'
pathlib.Path(base).mkdir(parents=True, exist_ok=True)
print('backup dir:', base)
PY

# 6) Treino forte pedido: 60 epochs e 14 seeds
# Ja salva resultado por seed/modo em outputs/result_<mode>_seed<seed>.json
# E gera log completo para retomar analise se cair.
cd /content/oars-colab-mvp
python scripts/run_ablation.py --epochs 60 --seeds 14 --samples 5000 2>&1 | tee outputs/train_60x14.log

# 7) Snapshot intermediario para Drive (apos treino)
python - << 'PY'
import shutil, pathlib
src = pathlib.Path('/content/oars-colab-mvp/outputs')
dst = pathlib.Path('/content/drive/MyDrive/oars_mvp_backup/outputs_last')
if dst.exists():
    shutil.rmtree(dst)
shutil.copytree(src, dst)
print('snapshot saved to', dst)
PY

# 8) Graficos + estatisticas finais
cd /content/oars-colab-mvp
python scripts/plot_results.py --csv outputs/ablation_summary.csv
cat outputs/ablation_stats.json

# 9) Ver rapidamente se tem dados intermediarios
ls outputs
ls outputs/plots
