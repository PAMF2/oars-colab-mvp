# Rode em UMA celula Python no Kaggle.

import subprocess
from pathlib import Path

repo = Path('/kaggle/working/oars-colab-mvp')
if repo.exists():
    subprocess.run('git -C /kaggle/working/oars-colab-mvp pull --ff-only', shell=True, check=True)
else:
    subprocess.run('git clone https://github.com/PAMF2/oars-colab-mvp /kaggle/working/oars-colab-mvp', shell=True, check=True)

subprocess.run("pip -q install -e /kaggle/working/oars-colab-mvp[dev]", shell=True, check=True)

# 1) baixar miniF2F raw
subprocess.run('python scripts/download_minif2f.py --out minif2f_raw.jsonl', shell=True, cwd=repo, check=True)

# 2) preparar e validar dataset de features
subprocess.run('python scripts/prepare_minif2f.py --input minif2f_raw.jsonl --output data/minif2f_prepared.jsonl --input-dim 80 --write-split', shell=True, cwd=repo, check=True)
subprocess.run('python scripts/validate_dataset.py --input data/minif2f_prepared.jsonl --input-dim 80', shell=True, cwd=repo, check=True)

# 3) treino real de modelo causal LM (SFT)
subprocess.run(
    'python scripts/train_real_model.py --data minif2f_raw.jsonl --model Qwen/Qwen2.5-0.5B --out-dir outputs/real_model --epochs 1 --batch-size 1 --grad-accum 16 --limit 2000',
    shell=True,
    cwd=repo,
    check=True,
)

# 4) avaliacao formal Lean pass@k
subprocess.run(
    'python scripts/run_putnam_passk.py --data minif2f_raw.jsonl --k 32 --verifier lean --require-lean --limit 200 --out outputs/putnam_passk_report_lean.json',
    shell=True,
    cwd=repo,
    check=True,
)

subprocess.run('python - << "PY"\nimport json\nfrom pathlib import Path\nr=json.loads(Path("outputs/putnam_passk_report_lean.json").read_text())\nprint(r)\nPY', shell=True, cwd=repo, check=True)
