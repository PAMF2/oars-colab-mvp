# COLAB - M2 V3.2 (ARLA CORRIGIDO) - BLOCO UNICO PYTHON
import subprocess
import pathlib

root = pathlib.Path('/content')
repo = root / 'oars-colab-mvp'

if repo.exists():
    subprocess.run('git pull', shell=True, cwd=repo, check=True)
else:
    subprocess.run('git clone https://github.com/PAMF2/oars-colab-mvp.git', shell=True, cwd=root, check=True)

subprocess.run('pip -q install -e .[dev] datasets huggingface_hub', shell=True, cwd=repo, check=True)

# baixar miniF2F real
subprocess.run(
    "python - << 'PY'\n"
    "from huggingface_hub import hf_hub_download\n"
    "import shutil\n"
    "src = hf_hub_download(repo_id='Tonic/MiniF2F', repo_type='dataset', filename='minif2f.jsonl')\n"
    "shutil.copy(src, 'minif2f_raw.jsonl')\n"
    "print('saved minif2f_raw.jsonl')\n"
    "PY",
    shell=True,
    cwd=repo,
    check=True,
)

# rodar M2 robusto
subprocess.run(
    'python scripts/run_m2.py --raw minif2f_raw.jsonl --kfold 5 --seeds 10 --epochs 20 --m2-threshold 0.01',
    shell=True,
    cwd=repo,
    check=True,
)

# mostrar resultado final
subprocess.run('python - << "PY"\nimport json\nfrom pathlib import Path\np=Path("outputs/m2/m2_report.json")\nr=json.loads(p.read_text())\nprint("m2_pass:",r["m2_pass"])\nprint("best_arla_gain:",r["gains_vs_baseline"]["best_arla_gain"])\nprint("arla_block_gain:",r["gains_vs_baseline"]["arla_block"])\nprint("arla_full_gain:",r["gains_vs_baseline"]["arla_full"])\nprint("recommendation:",r["recommendation"])\nprint("summary:",r["summary"])\nPY', shell=True, cwd=repo, check=True)
