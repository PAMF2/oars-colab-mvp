# COLAB - TREINAR ENCODER/DECODER E RODAR PHASE I COM ENCODER (BLOCO UNICO PYTHON)
import subprocess
import pathlib

root = pathlib.Path('/content')
repo = root / 'oars-colab-mvp'

if repo.exists():
    subprocess.run('git pull', shell=True, cwd=repo, check=True)
else:
    subprocess.run('git clone https://github.com/PAMF2/oars-colab-mvp.git', shell=True, cwd=root, check=True)

subprocess.run('pip -q install -e .[dev] datasets huggingface_hub', shell=True, cwd=repo, check=True)

# baixar miniF2F real
subprocess.run(
    "python - << 'PY'\n"
    "from huggingface_hub import hf_hub_download\n"
    "import shutil\n"
    "src = hf_hub_download(repo_id='Tonic/MiniF2F', repo_type='dataset', filename='minif2f.jsonl')\n"
    "shutil.copy(src, 'minif2f_raw.jsonl')\n"
    "print('saved minif2f_raw.jsonl')\n"
    "PY",
    shell=True,
    cwd=repo,
    check=True,
)

# preparar dataset de features
subprocess.run(
    'python scripts/prepare_minif2f.py --input minif2f_raw.jsonl --output data/minif2f_prepared.jsonl --input-dim 80 --write-split',
    shell=True,
    cwd=repo,
    check=True,
)

# treinar encoder/decoder
subprocess.run(
    'python scripts/train_autoencoder.py --data data/minif2f_prepared.jsonl --epochs 30 --latent-dim 32 --hidden-dim 128 --out-dir outputs/autoencoder',
    shell=True,
    cwd=repo,
    check=True,
)

# ativar encoder no config
subprocess.run(
    "python - << 'PY'\n"
    "import yaml\n"
    "p='configs/default.yaml'\n"
    "c=yaml.safe_load(open(p,'r',encoding='utf-8'))\n"
    "c['encoder']['use_pretrained']=True\n"
    "c['encoder']['checkpoint']='outputs/autoencoder/best_autoencoder.pt'\n"
    "yaml.safe_dump(c, open(p,'w',encoding='utf-8'), sort_keys=False)\n"
    "print(c['encoder'])\n"
    "PY",
    shell=True,
    cwd=repo,
    check=True,
)

# rodar milestones phase I
subprocess.run(
    'python scripts/run_phase1_roadmap.py --raw minif2f_raw.jsonl --seeds 10 --epochs 20 --out outputs/phase1/phase1_report.json',
    shell=True,
    cwd=repo,
    check=True,
)

# mostrar relatorio final
subprocess.run('cat outputs/phase1/phase1_report.json', shell=True, cwd=repo, check=True)
