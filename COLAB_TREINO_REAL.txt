# COLAB V3 (MINIMO 80 FEATURES) - PUXAR DO GITHUB E EXECUTAR
# BLOCO UNICO PYTHON
import subprocess
import pathlib

root = pathlib.Path('/content')
repo = root / 'oars-colab-mvp'

if repo.exists():
    subprocess.run('git pull', shell=True, cwd=repo, check=True)
else:
    subprocess.run('git clone https://github.com/PAMF2/oars-colab-mvp.git', shell=True, cwd=root, check=True)

subprocess.run('pip -q install -e .[dev] datasets huggingface_hub', shell=True, cwd=repo, check=True)

# baixar miniF2F real sem upload
subprocess.run(
    "python - << 'PY'\n"
    "from huggingface_hub import hf_hub_download\n"
    "import shutil\n"
    "src = hf_hub_download(repo_id='Tonic/MiniF2F', repo_type='dataset', filename='minif2f.jsonl')\n"
    "shutil.copy(src, 'minif2f_raw.jsonl')\n"
    "print('saved minif2f_raw.jsonl')\n"
    "PY",
    shell=True,
    cwd=repo,
    check=True,
)

# preparar V3: input_dim 80 + split train/val/test estratificado
subprocess.run(
    'python scripts/prepare_minif2f.py --input minif2f_raw.jsonl --output data/minif2f_prepared.jsonl --input-dim 80 --write-split',
    shell=True,
    cwd=repo,
    check=True,
)

# rodar ablation v3 (multiclasse + macro-f1)
subprocess.run('python scripts/run_ablation.py --epochs 12 --seeds 30 --samples 2000', shell=True, cwd=repo, check=True)
subprocess.run('python scripts/plot_results.py --csv outputs/ablation_summary.csv', shell=True, cwd=repo, check=True)
subprocess.run('cat outputs/ablation_stats.json', shell=True, cwd=repo, check=True)
