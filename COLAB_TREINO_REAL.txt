# RETRY PASS@K REAL COM DIAGNOSTICO LEAN (BLOCO UNICO PYTHON)
import os
import subprocess
import pathlib

repo = pathlib.Path('/content/oars-colab-mvp')

# 1) atualizar repo
subprocess.run('git pull', shell=True, cwd=repo, check=True)

# 2) garantir PATH para elan
elan_bin = os.path.expanduser('~/.elan/bin')
os.environ['PATH'] = elan_bin + ':' + os.environ.get('PATH', '')

# 3) diagnostico lean
print('\n== LEAN DIAGNOSTICO ==')
subprocess.run('which lean || true', shell=True, check=False)
subprocess.run('ls -la ~/.elan/bin || true', shell=True, check=False)
subprocess.run('~/.elan/bin/lean --version || true', shell=True, check=False)
subprocess.run('lean --version || true', shell=True, check=False)

# 4) rodar pass@k real
subprocess.run(
    'python scripts/run_putnam_passk.py --data minif2f_raw.jsonl --k 32 --verifier lean --require-lean --limit 200 --out outputs/putnam_passk_report_lean.json',
    shell=True,
    cwd=repo,
    check=True,
)

# 5) mostrar resultado
subprocess.run('cat outputs/putnam_passk_report_lean.json', shell=True, cwd=repo, check=True)
